<script>
/* ===== Tracking ===== */
let mouseMoves = 0;
let hoverCount = 0;
let startTime = performance.now();
let revealIndex = 0;

document.addEventListener("mousemove", () => {
  mouseMoves++;
});

document.addEventListener("paste", e => {
  e.preventDefault();
  alert("Paste disabled.");
});

/* ===== Progressive Reveal ===== */
const words = document.querySelectorAll(".word");

words.forEach((word, index) => {
  word.addEventListener("mouseenter", () => {
    if (index === revealIndex) {
      word.classList.add("active");
      revealIndex++;
      hoverCount++;
    }
  });
});

/* ===== Submission ===== */
function submitForm() {
  const answer = document.getElementById("answer").value.trim();
  const timeSpent = Math.round((performance.now() - startTime) / 1000);

  let riskScore = 0;

  if (mouseMoves < 100) riskScore += 0.4;
  if (hoverCount < words.length) riskScore += 0.3;
  if (answer.length > 150 && timeSpent < 30) riskScore += 0.4;

  const submission = {
    answer: answer,
    timeSpentSeconds: timeSpent,
    mouseMoves: mouseMoves,
    hoverCount: hoverCount,
    riskScore: Math.min(riskScore, 1.0)
  };

  document.getElementById("result").classList.remove("hidden");
  document.getElementById("result").textContent =
    "SUBMISSION DATA (for review):\n\n" +
    JSON.stringify(submission, null, 2);
}
</script>
